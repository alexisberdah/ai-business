const [step, setStep] = useState(1)
const totalSteps = 4

const progress = (step / totalSteps) * 100

return (
  <Card>
    <Progress value={progress} />
    {step === 1 && <WelcomeStep />}
    {step === 2 && <ProfileStep />}
    {step === 3 && <PreferencesStep />}
    {step === 4 && <CompleteStep />}
  </Card>
